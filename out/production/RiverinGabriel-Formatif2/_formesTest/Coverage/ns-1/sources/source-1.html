 package formes;
 
 import exceptions.FormeException;
 
 import java.util.Objects;
 
 public class Cercle {
 
     private String couleur;
     public static final int RAYON_DEFAUT = 10;
     private static String nom = "Cercle";
     public static final String COULEUR_DEFAUT = "vert";
     public static final int MIN_VAL = 1;
     private int rayon;
     public static final int MAX_VAL = 30;
     public static final String[] LES_COULEURS = {"rouge", "vert", "bleu", "jaune", "noir", "orange"};
 
     public Cercle(int rayon) {
         setRayon(rayon);
         setCouleur(couleur);
     }
 
     public Cercle() {
         rayon = RAYON_DEFAUT;
     }
 
     public String getCouleur() {
         return couleur;
     }
 
     public void setCouleur(String couleur) {
         if (couleurEstValide(couleur)) {
             this.couleur = couleur.toLowerCase();
         } else {
             this.couleur = COULEUR_DEFAUT;
         }
     }
 
     public static String getNom() {
         return nom;
     }
 
 
     public int getRayon() {
         return rayon;
     }
 
     public void setRayon(int rayon) {
         if (rayonEstValide(rayon)) {
             this.rayon = rayon;
         } else {
             throw new FormeException("Rayon invalide doit etre entre 1 et 30: " + rayon);
         }
     }
 
     public int calculerSurface() {
         return (int) (Math.PI * rayon * rayon);
     }
 
     @Override
     public boolean equals(Object o) {
         if (this == o) return true;
         if (o == null || getClass() != o.getClass()) return false;
         Cercle cercle = (Cercle) o;
         return calculerSurface() == cercle.calculerSurface() && couleur == cercle.couleur;
     }
 
     public boolean rayonEstValide(int rayon) {
         return rayon >= MIN_VAL && rayon <= MAX_VAL;
     }
 
     public boolean couleurEstValide(String couleur) {
         for (int i = 0; i < LES_COULEURS.length; i++) {
             if (LES_COULEURS[i].equalsIgnoreCase(couleur)) {
                 return true;
             }
         }
         return false;
     }
 
     public int calculerPerimetre() {
         return (int) (2 * Math.PI * rayon);
     }
 
     @Override
     public String toString() {
         return nom + " " + couleur + " " + rayon;
     }
 }
