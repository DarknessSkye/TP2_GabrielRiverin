 package formes;
 
 import exceptions.FormeException;
 
 public class Rectangle {
 
     private String couleur;
     public static final String COULEUR_DEFAUT = "rouge";
     private int largeur;
     public static final int MIN_VAL = 1;
     private static String nom = "Rectangle";
     public static final int MAX_VAL = 30;
     public static final String[] LES_COULEURS = {"rouge", "vert", "bleu", "jaune", "noir", "orange"};
     private int hauteur;
 
 
     public Rectangle(int largeur, int hauteur) {
         setCouleur(couleur);
         setHauteur(hauteur);
         setLargeur(largeur);
     }
 
     public void setCouleur(String couleur) {
         if (couleurEstValide(couleur)) {
             this.couleur = couleur.toLowerCase();
         } else {
             this.couleur = COULEUR_DEFAUT;
         }
     }
 
     public void setLargeur(int largeur) {
         if (largeurEstValide(largeur)) {
             this.largeur = largeur;
         } else {
             throw new FormeException("largeur invalide doit etre entre 1 et 30: " + largeur);
         }
     }
 
     public void setHauteur(int hauteur) {
         if (valideHauteur(hauteur)) {
             this.hauteur = hauteur;
         } else {
             throw new FormeException("hauteur invalide doit etre entre 1 et 30: " + hauteur);
         }
     }
 
     public String getCouleur() {
         return couleur;
     }
 
     public int getLargeur() {
         return largeur;
     }
 
     public String getNom() {
         return nom;
     }
 
     public int getHauteur() {
         return hauteur;
     }
 
     public int calculerSurface() {
         return (hauteur * largeur);
     }
 
     public boolean largeurEstValide(int largeur) {
         return largeur >= MIN_VAL && largeur <= MAX_VAL;
     }
 
     private boolean valideHauteur(int hauteur) {
         return hauteur >= MIN_VAL && hauteur <= MAX_VAL;
     }
 
     public int calculerPerimetre() {
         return ((2 * hauteur) + (2 * largeur));
     }
 
     public boolean couleurEstValide(String couleur) {
         for (int i = 0; i < LES_COULEURS.length; i++) {
             if (LES_COULEURS[i].equalsIgnoreCase(couleur)) {
                 return true;
             }
         }
         return false;
     }
 
     @Override
     public String toString() {
         return nom + " " + couleur + " " + hauteur + ", " + largeur;
     }
 
     @Override
     public boolean equals(Object o) {
         if (this == o) return true;
         if (o == null || getClass() != o.getClass()) return false;
         Rectangle rectangle = (Rectangle) o;
         return calculerSurface() == rectangle.calculerSurface() && couleur == rectangle.couleur;
     }
 
 }
